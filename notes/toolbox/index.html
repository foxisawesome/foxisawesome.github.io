ffmpeg

  1. remove audio, subtitle track

for x in ./*.mkv; ffmpeg -i $x -map 0 -map -0: a:0 -map -0:s:0 -c copy ./out/$x.mkv

Note:

  • -map 0 selects all streams from the input.
  • -map -0:a then deselects all audio streams from the input.
  • -map 0: a:0 remove #0 streams from the input.
  • -map -0:s:0 remove #0 subtitle streams from the input.
  1. encode to apple HEVC

for x in ./*.mov; ffmpeg -i $x -c:v libx265 -crf 22 -c:a copy -tag:v hvc1 $x.mp4

ffmpeg -i input.mt2s -scodec copy -acodec copy -vcodec copy -f matroska input.mkv

mkvmerge

for x in ./*.mkv; mkvmerge -o ./out/$x.mkv -A $x.mp4 -D -a 1,3 $x

-A: no audio; -D: no video; -a 1,3: choose audio stream 1 and 3
