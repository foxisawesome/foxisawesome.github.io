18 K Sum

  2020-09-03   algo   leetcode  Comments

LC 18 4Sum

Given an array nums of n integers and an integer target, are there elements a, b, c, and d in nums such that a + b + c + d = target? Find all unique quadruplets in the array which gives the sum of target.

Note: The solution set must not contain duplicate quadruplets.

Solution

2020-09-03 Notes:

K - sum algo:

  1. sort nums
  2. recursion top down, reduce to a 2-Sum problem
  3. two pointers: l, r to find target

Note: to skip duplicates: if not (i > 0 and nums[i]==nums[i-1]):

class Solution:
    def fourSum(self, nums: List[int], target: int) -> List[List[int]]:

        nums.sort()
        return self.kSum(nums, target, 4)
    
    def kSum(self, nums: List[int], target: int, k: int) -> List[List[int]]:
        res = []
        if len(nums)==0 or nums[0] * k > target or nums[-1] * k < target:
            return res
        if k == 2:
            return self.twoSum(nums, target)

        for i in range(len(nums)):
            if i == 0 or nums[i - 1] != nums[i]:
                for _, val in enumerate(self.kSum(nums[i+1:], target-nums[i], k-1)):
                    res.append([nums[i]] + val)
        return res

    def twoSum(self, nums: List[int], target: int) -> List[List[int]]:
        res = []
        lo, hi = 0, len(nums) - 1
        while (lo < hi):
            sum = nums[lo] + nums[hi]
            if sum < target or (lo > 0 and nums[lo] == nums[lo - 1]):
                lo += 1
            elif sum > target or (hi < len(nums) - 1 and nums[hi] == nums[hi + 1]):
                hi -= 1
            else:
                res.append([nums[lo], nums[hi]])
                lo += 1
                hi -= 1
        return res

# Time O(N^(k-1))
# Space O(N)
